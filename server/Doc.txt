1. Auth
    POST /auth/register - registering User
    POST /auth/login - login User
2. User
    GET /user/:id - User details
    GET /user/:id/friends - user friends
    PATCH /user/:friendID - Add / remove friends
3. Post
    GET /post - All post
    GET /post/:userId - specific user posts By UserID
    GET /post/:postId - specific post By ID
    POST /post - create post
    PATCH /post/like/:postID - like / dislike fun


Schema
1. User
    firstName: String
    lastName: String
    email: String
    password: String
    picturePath: String
    friends: Array of UserID
    location: String
    occupation: String

2. Post
    userId: String
    location: String
    description: String
    picturePath: String
    likes: Array of UserID
    comments: Array of userID + Msg
